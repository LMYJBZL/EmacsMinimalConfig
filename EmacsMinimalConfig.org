(defun minimal-emacs--remove-el-file-suffix (filename)
  "Remove the Elisp file suffix from FILENAME and return it (.el, .el.gz...)."
  (let ((suffixes (mapcar (lambda (ext) (concat ".el" ext))
                          load-file-rep-suffixes)))
    (catch 'done
      (dolist (suffix suffixes filename)
        (when (string-suffix-p suffix filename)
          (setq filename (substring filename 0 (- (length suffix))))
          (throw 'done t))))
    filename))

#+begin_src emacs-lisp
;; Garbage collection significantly affects startup times. This setting delays
;; garbage collection during startup but will be reset later.

(defun minimal-emacs--restore-gc-values ()
  "Restore garbage collection values to minimal-emacs.d values."
  (setq gc-cons-threshold minimal-emacs-gc-cons-threshold)
  (setq gc-cons-percentage minimal-emacs-gc-cons-percentage))

(defun minimal-emacs--restore-gc ()
  "Restore garbage collection settings."
  (if (and (bound-and-true-p minimal-emacs-gc-cons-threshold-restore-delay)
           ;; In noninteractive mode, the event loop does not run
           (not noninteractive))
      ;; Defer garbage collection during initialization to avoid 2 collections.
      (run-with-timer minimal-emacs-gc-cons-threshold-restore-delay nil
                      #'minimal-emacs--restore-gc-values)
    (minimal-emacs--restore-gc-values)))

(if minimal-emacs-optimize-startup-gc
    ;; `gc-cons-threshold' is managed by minimal-emacs.d
    (add-hook 'emacs-startup-hook #'minimal-emacs--restore-gc 105)
  ;; gc-cons-threshold is not managed by minimal-emacs.d.
  (when (= gc-cons-threshold most-positive-fixnum)
    (setq gc-cons-threshold minimal-emacs--backup-gc-cons-threshold)
    (setq gc-cons-percentage minimal-emacs--backup-gc-cons-percentage)))

#+begin_src emacs-lisp
(defvar minimal-emacs--old-file-name-handler-alist (default-toplevel-value
                                                    'file-name-handler-alist))

(defun minimal-emacs--respect-file-handlers (fn args-left)
  "Respect file handlers.
FN is the function and ARGS-LEFT is the same argument as `command-line-1'.
Emacs processes command-line files very early in startup. These files may
include special paths like TRAMP paths, so restore `file-name-handler-alist' for
this stage of initialization."
  (let ((file-name-handler-alist (if args-left
                                     minimal-emacs--old-file-name-handler-alist
                                   file-name-handler-alist)))
    (funcall fn args-left)))

(defun minimal-emacs--restore-file-name-handler-alist ()
  "Restore `file-name-handler-alist'."
  (set-default-toplevel-value
   'file-name-handler-alist
   ;; Merge instead of overwrite to preserve any changes made since startup.
   (delete-dups (append file-name-handler-alist
                        minimal-emacs--old-file-name-handler-alist))))

(when (and minimal-emacs-optimize-file-name-handler-alist
           (not minimal-emacs-debug))
  ;; Determine the state of bundled libraries using calc-loaddefs.el. If
  ;; compressed, retain the gzip handler in `file-name-handler-alist`. If
  ;; compiled or neither, omit the gzip handler during startup for improved
  ;; startup and package load time.
  (set-default-toplevel-value
   'file-name-handler-alist
   (if (locate-file-internal "calc-loaddefs.el" load-path)
       nil
     (list (rassq 'jka-compr-handler
                  minimal-emacs--old-file-name-handler-alist))))

  ;; Ensure the new value persists through any current let-binding.
  (put 'file-name-handler-alist 'initial-value
       minimal-emacs--old-file-name-handler-alist)

  ;; Emacs processes command-line files very early in startup. These files may
  ;; include special paths TRAMP. Restore `file-name-handler-alist'.
  (advice-add 'command-line-1 :around #'minimal-emacs--respect-file-handlers)

  (add-hook 'emacs-startup-hook #'minimal-emacs--restore-file-name-handler-alist
            101))

(defun minimal-emacs--startup-load-user-init-file (fn &rest args)
  "Advice to reset `mode-line-format'. FN and ARGS are the function and args."
  (unwind-protect
      ;; Start up as normal
      (apply fn args)
    ;; If we don't undo inhibit-{message, redisplay} and there's an error, we'll
    ;; see nothing but a blank Emacs frame.
    (when minimal-emacs-inhibit-message-during-startup
      (setq-default inhibit-message nil))
    (when minimal-emacs-inhibit-redisplay-during-startup
      (setq-default inhibit-redisplay nil))
    ;; Restore the mode-line
    (when minimal-emacs-disable-mode-line-during-startup
      (unless (default-toplevel-value 'mode-line-format)
        (setq-default mode-line-format (get 'mode-line-format
                                            'initial-value)))
      (dolist (buf (buffer-list))
        (with-current-buffer buf
          (when (local-variable-p 'minimal-emacs--hidden-mode-line)
            (setq mode-line-format minimal-emacs--hidden-mode-line)
            (kill-local-variable 'minimal-emacs--hidden-mode-line)))))))

#+begin_src emacs-lisp
;; Native compilation enhances Emacs performance by converting Elisp code into
;; native machine code, resulting in faster execution and improved
;; responsiveness.
;;
;; Ensure adding the following compile-angel code at the very beginning
;; of your `~/.emacs.d/post-init.el` file, before all other packages.
(use-package compile-angel
  :demand t
  :config
  ;; The following disables compilation of packages during installation;
  ;; compile-angel will handle it.
  (setq package-native-compile nil)

  ;; Set `compile-angel-verbose' to nil to disable compile-angel messages.
  ;; (When set to nil, compile-angel won't show which file is being compiled.)
  (setq compile-angel-verbose t)

  ;; The following directive prevents compile-angel from compiling your init
  ;; files. If you choose to remove this push to `compile-angel-excluded-files'
  ;; and compile your pre/post-init files, ensure you understand the
  ;; implications and thoroughly test your code. For example, if you're using
  ;; the `use-package' macro, you'll need to explicitly add:
  ;; (eval-when-compile (require 'use-package))
  ;; at the top of your init file.

** Magit

magit是Emacs的一个强大的git前端.

工作流程如下：
1. 修改本地仓储
2. stage这些修改
3. 提交这些修改
4. push到远端github

一个最小的 magit 流程包括：

C-x g 打开 magit buffer 查看仓储状态

要暂存或取消暂存项目, 您可以键入 s 或 u 来暂存/取消暂存该项目（无论是整个文件还是一个代码块）

如果要删除某些内容, 可以使用 k 命令.
它适用于暂存文件、未暂存文件、代码块和未跟踪文件.

要在 Magit 中开始提交更改, 请键入 c.
此时会弹出一个窗口——该弹出窗口是您与 Magit 交互并修改其必须传递给 Git 的参数的主要方式.
Magit 可能会添加一些在 Git 中不存在的命令或概念.
这些操作仅存在于 Magit 中.

Magit 的魅力之一在于您可以轻松地自定义每个命令.
但如果您不需要自定义命令, 只需再次按下 c 即可打开提交消息缓冲区.

M-n 和 M-p

它们用于循环浏览提交信息历史记录.
因此, 如果您编写了一条很长的提交信息, 但在提交之前意识到需要修改一些已暂存的更改, 可以通过按 M-p 来恢复您的草稿提交信息.

C-c C-c

提交包含您编写的信息的更改.

Magit 会检查您的信息是否过长, 并在您违反“不成文”的规则（即 Git 提交信息的第一行必须包含完整的内容且长度不超过 70 个字符左右）时发出警告.
（这样，当您使用 `git log --one-line` 命令时, 信息就不会被截断.）

此限制不适用于您编写的后续行或段落.（但如果您尚未遵守此规则，我建议您遵守）

C-c C-k

取消提交. 您将返回到 Magit 窗口, 如果需要，可以通过重复按 c c 组合键来恢复提交.

create a new repository on the command line

echo "# EmacsMinimalConfig" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin git@github.com:LMYJBZL/EmacsMinimalConfig.git
git push -u origin main

push an existing repository from the command line

git remote add origin git@github.com:LMYJBZL/EmacsMinimalConfig.git
git branch -M main

(defun my/latex-tab ()
  "Smart Tab for LaTeX:
1. indent
2. yasnippet expand/next field
3. cdlatex-style cursor jump"
  (interactive)
  (let ((orig-point (point)))
    ;; 正常缩进
    (indent-for-tab-command)
    ;; snippet 展开或跳转
    (when (= orig-point (point))
      (unless (yas-expand)
        (when (yas--snippets-at-point)
          (yas-next-field))))
    ;;  cdlatex 风格跳转
    (when (= orig-point (point))
      (catch 'stop
        ;; 初始前进一步（避免死循环）
        (forward-char 1)
        (while (re-search-forward "[ )}\n]\\|\\]" (point-max) t)
          (forward-char -1)
          (cond
           ;; 遇到空格 → 停在第一个空格（或行首）
           ((= (following-char) ?\ )
            (if (not (bolp)) (forward-char 1))
            (throw 'stop t))
           ;; 遇到换行 → 停在行首/行尾（但 \\ 例外）
           ((= (following-char) ?\n)
            (if (and (bolp) (not (eobp)))
                (throw 'stop t)
              (if (equal "\\\\" (buffer-substring-no-properties
                                 (- (point) 2) (point)))
                  (forward-char 1)
                (throw 'stop t))))
           ;; 遇到右括号 → 根据前面字符判断停前还是停后
           (t
            (if (or (= (char-syntax (preceding-char)) ?\()
                    (= (char-syntax (preceding-char)) ?\))
                    (= (preceding-char) ?-))
                (throw 'stop t)
              (forward-char 1)
              (if (looking-at "[^_^({[]")
                  (throw 'stop t))))))))))

(defconst latex-basic-braces
  '(("(" . ")")
    ("[" . "]")
    ("\\{" . "\\}")
    ("|" . "|")
    ("\\|" . "\\|")
    ("\\lfloor" . "\\rfloor")
    ("\\lceil" . "\\rceil")
    ("\\langle" . "\\rangle")
    ("\\backslash" . "\\backslash"))
  "基本 LaTeX 括号符号表")

(defun latex-generate-delim-pairs ()
  "生成所有层次的 LaTeX 括号配对，包括 \\left, \\big, \\Big, \\bigg, \\Bigg 等。"
  (let ((prefixes '("" "\\left" "\\bigl" "\\Bigl" "\\biggl" "\\Biggl"))
        (suffixes '("" "\\right" "\\bigr" "\\Bigr" "\\biggr" "\\Biggr"))
        result)
    (dolist (pre prefixes)
      (dolist (suf suffixes)
        (dolist (pair latex-basic-braces)
          (push (cons (concat pre (car pair)) (concat suf (cdr pair))) result))))
    ;; 添加非闭合情况
    (dolist (left latex-basic-braces)
      (push (cons (concat "\\left" (car left)) "\\right.") result)
      (push (cons "\\left." (concat "\\right" (cdr left))) result))
    (nreverse result)))

;;; ------------------------------
(defvar latex-brace-pairs
  (append
   ;; 普通括号
   '(("\\(" . "\\)") ("\\[" . "\\]") ("\\{" . "\\}"))
   ;; 自动生成的所有 left/big 系列
   (latex-generate-delim-pairs)))

;; 预排序以提升性能（按左/右项长度降序）
(defvar latex-brace-pairs-left-sorted
  (sort (copy-sequence latex-brace-pairs)
        (lambda (a b) (> (length (car a)) (length (car b))))))
(defvar latex-brace-pairs-right-sorted
  (sort (copy-sequence latex-brace-pairs)
        (lambda (a b) (> (length (cdr a)) (length (cdr b))))))

;;; ------------------------------
(defun my/TeX--blank-between (start end)
  "检查 START 与 END 之间是否只有空白字符。"
  (string-blank-p (buffer-substring-no-properties start end)))

(defun my/TeX--find-matching-right (pos left-symbol)
  "从 POS (光标位置) 向后查找与 LEFT-SYMBOL 匹配的右符号。
返回 (RIGHT-POS RIGHT-LEN) —— RIGHT-POS 为 match 结束位置（search-forward 返回的点），或 nil。
搜索范围限制在 pos 后 5000 字符内，避免大文件卡顿。"
  (let* ((rights (mapcar #'cdr
                         (cl-remove-if-not
                          (lambda (p) (string= (car p) left-symbol))
                          latex-brace-pairs)))
         (sorted-rights (sort (copy-sequence rights)
                              (lambda (a b) (> (length a) (length b)))))
         ;; 限制搜索范围到 5000 字符内
         (limit (min (point-max) (+ pos 5000))))
    (catch 'found
      (dolist (r sorted-rights)
        (save-excursion
          (goto-char pos)
          ;; 添加 limit 参数，防止搜索到文件末尾
          (let ((rpos (search-forward r limit t)))
            (when rpos
              (throw 'found (list rpos (length r)))))))
      nil)))

(defun my/TeX--find-matching-left (pos right-symbol)
  "从 POS (光标位置) 向前查找与 RIGHT-SYMBOL 匹配的左符号。
返回 (LEFT-POS LEFT-LEN) —— LEFT-POS 为 match 开始位置，或 nil。
搜索范围限制在 pos 前 5000 字符内，避免大文件卡顿。"
  (let* ((lefts (mapcar #'car
                        (cl-remove-if-not
                         (lambda (p) (string= (cdr p) right-symbol))
                         latex-brace-pairs)))
         (sorted-lefts (sort (copy-sequence lefts)
                             (lambda (a b) (> (length a) (length b)))))
         ;; 限制搜索范围，向前搜索到 pos 前 5000 字符
         (limit (max (point-min) (- pos 5000))))
    (catch 'found
      (dolist (l sorted-lefts)
        (save-excursion
          ;; 从 right 符号开始位置向前搜索左符号
          (goto-char (- pos (length right-symbol)))
          ;; 添加 limit 参数，防止搜索到文件开头
          (let ((lpos (search-backward l limit t)))
            (when lpos
              (throw 'found (list lpos (length l)))))))
      nil)))

(defun my/TeX--delete-left-match (pos)
  "如果光标在某个左符号之后，尝试删除该左符号与其右配对（若中间为空白则删除整对），否则只删除左符号。
返回 t 表示已做删除，nil 表示未处理。"
  (catch 'done
    (dolist (pair latex-brace-pairs-left-sorted)
      (let* ((left (car pair))
             (llen (length left)))
        (when (and (>= pos llen)
                   (string= (buffer-substring-no-properties (- pos llen) pos) left))
          ;; 找到左符号匹配
          (let ((match (my/TeX--find-matching-right pos left)))
            (if match
                (let* ((rpos (nth 0 match))
                       (rlen (nth 1 match))
                       (content-start pos)
                       (content-end (- rpos rlen)))
                  (if (<= content-start content-end)
                      (if (my/TeX--blank-between content-start content-end)
                          (progn
                            (delete-region (- pos llen) rpos)
                            (throw 'done t))
                        ;; 中间非空白，只删除左符号
                        (delete-region (- pos llen) pos)
                        (throw 'done t))
                    ;; 如果 content-start > content-end（不应发生），退回删除左符号
                    (delete-region (- pos llen) pos)
                    (throw 'done t)))
              ;; 没有找到右匹配，仅删除左符号
              (delete-region (- pos llen) pos)
              (throw 'done t))))))
    nil))

(defun my/TeX--delete-right-match (pos)
  "如果光标在某个右符号之后，尝试删除该右符号与其左配对（若中间为空白则删除整对），否则只删除右符号。
返回 t 表示已做删除，nil 表示未处理。"
  (catch 'done
    (dolist (pair latex-brace-pairs-right-sorted)
      (let* ((right (cdr pair))
             (rlen (length right)))
        (when (and (>= pos rlen)
                   (string= (buffer-substring-no-properties (- pos rlen) pos) right))
          ;; 找到右符号匹配
          (let ((match (my/TeX--find-matching-left pos right)))
            (if match
                (let* ((lpos (nth 0 match))
                       (llen (nth 1 match))
                       (content-start (+ lpos llen))
                       (content-end (- pos rlen)))
                  (if (<= content-start content-end)
                      (if (my/TeX--blank-between content-start content-end)
                          (progn
                            (delete-region lpos pos)
                            (throw 'done t))
                        ;; 中间非空白，只删除右符号
                        (delete-region (- pos rlen) pos)
                        (throw 'done t))
                    ;; content-start > content-end 的回退行为
                    (delete-region (- pos rlen) pos)
                    (throw 'done t)))
              ;; 没有找到左匹配，仅删除右符号
              (delete-region (- pos rlen) pos)
              (throw 'done t))))))
    nil))

(defun my/TeX--delete-pair ()
  "智能删除 LaTeX 成对符号或单个符号。优先尝试以成对删除（当中间仅空白时），否则删除单个符号。
回退：若无匹配，行为与 `backward-delete-char-untabify` 相同。"
  (interactive)
  (unless (or (my/TeX--delete-left-match (point))
              (my/TeX--delete-right-match (point)))
    (call-interactively 'backward-delete-char-untabify)))

(defcustom my-auctex-auto-engine-chinese-tex-regexp
  (concat
   "^[[:space:]]*\\\\documentclass"
   "\\(?:\\[[^]]*\\]\\)?"
   "{\\(ctexart\\|ctexrep\\|ctexbook\\|ctexbeamer\\)}"
   "\\|^[[:space:]]*\\\\usepackage"
   "\\(?:\\[[^]]*\\]\\)?"
   "{\\([^}]*,\\)?ctex\\(,[^}]*\\)?}")
  "Regexp used to decide whether the current TeX buffer uses Chinese TeX support."
  :type 'regexp
  :group 'my-auctex-auto-engine)

(defcustom my-auctex-auto-engine-local-vars-prefix "%%%"
  "Comment prefix used when writing the file-local variable block."
  :type 'string
  :group 'my-auctex-auto-engine)

(defun my-auctex--file-local-variables-bounds ()
  "Return the bounds of the file-local variable block, or nil."
  (save-excursion
    (goto-char (point-max))
    (let (start end)
      (when (re-search-backward "^[[:space:]]*\\(?:[%#;]+[[:space:]]*\\)?Local Variables:[[:space:]]*$" nil t)
        (setq start (line-beginning-position))
        (when (re-search-forward "^[[:space:]]*\\(?:[%#;]+[[:space:]]*\\)?End:[[:space:]]*$" nil t)
          (setq end (line-end-position))
          (cons start end))))))

(defun my-auctex--file-local-variables-text ()
  "Return the file-local variable block text for the current buffer, or nil."
  (let ((bounds (my-auctex--file-local-variables-bounds)))
    (when bounds
      (buffer-substring-no-properties (car bounds) (cdr bounds)))))

(defun my-auctex--buffer-uses-chinese-tex-p ()
  "Return non-nil when the current buffer uses Chinese TeX classes or packages."
  (save-excursion
    (goto-char (point-min))
    (re-search-forward my-auctex-auto-engine-chinese-tex-regexp nil t)))

(defun my-auctex--desired-engine ()
  "Return the TeX engine that matches the current buffer content."
  (if (my-auctex--buffer-uses-chinese-tex-p) 'xetex 'luatex))

   #'identity
   (list ""

         (format "%s End:" my-auctex-auto-engine-local-vars-prefix))
   "\n"))

(defun my-auctex--write-engine-to-local-variables (engine)
  "Write ENGINE into the file-local variable block for the current buffer."
  (let ((bounds (my-auctex--file-local-variables-bounds))
        (engine-line (my-auctex--local-vars-engine-line engine)))
    (save-excursion
      (save-restriction
        (widen)
        (cond
         (bounds
          (goto-char (car bounds))
          (let ((block-end-marker (copy-marker (cdr bounds) t)))
            (if (re-search-forward "^[[:space:]]*\\(?:[%#;]+[[:space:]]*\\)?TeX-engine:[[:space:]]*.*$" block-end-marker t)
                (replace-match engine-line t t)
              (goto-char block-end-marker)
              (forward-line -1)
              (end-of-line)
              (insert "\n" engine-line))))
         (t
          (goto-char (point-max))
          (insert (my-auctex--local-vars-block engine))))))))

(define-advice keyboard-quit
    (:around (quit) quit-current-context)
  "Quit the current context.

When there is an active minibuffer and we are not inside it close
it.  When we are inside the minibuffer use the regular
`minibuffer-keyboard-quit' which quits any active region before
exiting.  When there is no minibuffer `keyboard-quit' unless we
are defining or executing a macro."
  (if (active-minibuffer-window)
      (if (minibufferp)
          (minibuffer-keyboard-quit)
        (abort-recursive-edit))
    (unless (or defining-kbd-macro
                executing-kbd-macro)
      (funcall-interactively quit))))
